import React, { useState } from "react";
import { motion, AnimatePresence } from "framer-motion";
import { useCreateLoadstring } from "@workspace/api-client-react";
import { 
  Terminal, 
  Zap, 
  Copy, 
  Check, 
  ExternalLink, 
  Loader2, 
  AlertCircle 
} from "lucide-react";
import { Button } from "@/components/ui/button";

export default function Home() {
  const [code, setCode] = useState("");
  const [copied, setCopied] = useState(false);
  const { mutate, isPending, data, error, isSuccess, reset } = useCreateLoadstring();

  const handleCreate = () => {
    if (!code.trim()) return;
    mutate({ data: { code } });
  };

  const handleCopy = () => {
    if (data?.loadstring) {
      navigator.clipboard.writeText(data.loadstring);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    }
  };

  return (
    <div className="min-h-[100dvh] w-full flex flex-col items-center justify-start pt-12 pb-24 px-4 sm:px-6 relative overflow-x-hidden">
      
      {/* Header */}
      <motion.header 
        initial={{ opacity: 0, y: -20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.6, ease: "easeOut" }}
        className="w-full max-w-4xl flex items-center justify-between mb-12"
      >
        <div className="flex items-center gap-3">
          <div className="w-10 h-10 rounded-xl bg-primary/10 border border-primary/30 flex items-center justify-center text-primary">
            <Terminal className="w-5 h-5" />
          </div>
          <div>
            <h1 className="text-xl font-bold tracking-tight text-primary">
              LoadString Creations
            </h1>
            <p className="text-xs text-muted-foreground font-mono">
              SECURE // FAST // ANONYMOUS
            </p>
          </div>
        </div>
      </motion.header>

      {/* Main Content */}
      <main className="w-full max-w-4xl flex flex-col gap-8">
        
        {/* Input Section */}
        <motion.section 
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ duration: 0.6, delay: 0.1, ease: "easeOut" }}
          className="flex flex-col gap-4"
        >
          <div className="flex items-center justify-between px-2">
            <h2 className="text-sm font-semibold tracking-wider uppercase text-muted-foreground flex items-center gap-2">
              <Zap className="w-4 h-4 text-primary" />
              Source Script
            </h2>
            {code.length > 0 && (
              <span className="text-xs font-mono text-muted-foreground/60">
                {code.length} chars
              </span>
            )}
          </div>
          
          <textarea
            value={code}
            onChange={(e) => { setCode(e.target.value); if (error) reset(); }}
            placeholder="-- Paste your raw Lua script here..."
            className="w-full h-[400px] resize-y bg-[#050508] border border-border rounded-xl p-6 font-mono text-sm text-foreground placeholder:text-muted-foreground/30 focus:outline-none focus:ring-1 focus:ring-primary focus:border-primary transition-all"
            spellCheck={false}
          />

          <div className="flex items-center justify-between mt-2">
            <div className="flex-1">
              <AnimatePresence>
                {error && (
                  <motion.div
                    initial={{ opacity: 0, x: -10 }}
                    animate={{ opacity: 1, x: 0 }}
                    exit={{ opacity: 0 }}
                    className="flex items-center gap-2 text-destructive text-sm font-medium"
                  >
                    <AlertCircle className="w-4 h-4" />
                    <span>{error.error || "Failed to create loadstring"}</span>
                  </motion.div>
                )}
              </AnimatePresence>
            </div>

            <Button
              size="lg"
              onClick={handleCreate}
              disabled={isPending || !code.trim()}
              className="bg-primary hover:bg-primary/90 text-primary-foreground font-bold tracking-wide rounded-lg px-8 h-12"
            >
              {isPending ? (
                <><Loader2 className="w-5 h-5 animate-spin" /> ENCRYPTING...</>
              ) : (
                <><Terminal className="w-5 h-5" /> CREATE LOADSTRING</>
              )}
            </Button>
          </div>
        </motion.section>

        {/* Result Section */}
        <AnimatePresence>
          {isSuccess && data && (
            <motion.section
              initial={{ opacity: 0, height: 0, y: 20 }}
              animate={{ opacity: 1, height: "auto", y: 0 }}
              transition={{ duration: 0.5, type: "spring", bounce: 0.3 }}
              className="flex flex-col gap-4 pt-8 border-t border-border/50"
            >
              <h2 className="text-sm font-semibold tracking-wider uppercase text-primary flex items-center gap-2 px-2">
                <Check className="w-4 h-4" />
                Loadstring Generated
              </h2>

              <div className="bg-[#050508] border border-primary/30 rounded-xl p-1">
                <div className="flex flex-col sm:flex-row items-stretch gap-2 bg-[#0a0a0f] rounded-lg p-2">
                  <div className="flex-1 overflow-x-auto p-4 flex items-center">
                    <code className="font-mono text-sm text-primary whitespace-nowrap">
                      {data.loadstring}
                    </code>
                  </div>
                  
                  <Button
                    variant="secondary"
                    size="lg"
                    onClick={handleCopy}
                    className="shrink-0 border border-primary/20 hover:border-primary/50 hover:bg-primary/10"
                  >
                    {copied ? (
                      <><Check className="w-4 h-4 text-green-400" /><span className="text-green-400">COPIED</span></>
                    ) : (
                      <><Copy className="w-4 h-4" /> COPY</>
                    )}
                  </Button>
                </div>
              </div>

              <div className="flex justify-end px-2">
                <a 
                  href={data.pasteUrl}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="flex items-center gap-1.5 text-xs font-mono text-muted-foreground hover:text-primary transition-colors"
                >
                  <span>View Raw Paste</span>
                  <ExternalLink className="w-3 h-3" />
                </a>
              </div>
            </motion.section>
          )}
        </AnimatePresence>

      </main>
    </div>
  );
}
